STEP 3: First Jarvis Code (RUN THIS 🔥)

Create a file: jarvis.py

python
import speech_recognition as sr
import pyttsx3

# Initialize voice engine
engine = pyttsx3.init()

def speak(text):
    engine.say(text)
    engine.runAndWait()

def listen():
    r = sr.Recognizer()
    with sr.Microphone() as source:
        print("Listening...")
        audio = r.listen(source)
    try:
        command = r.recognize_google(audio)
        print("You said:", command)
        return command.lower()
    except:
        return ""

speak("Hello Shweta, I am Jarvis. How can I help you?")

while True:
    command = listen()

    if "hello" in command:
        speak("Hello! Nice to talk to you.")
    elif "time" in command:
        import datetime
        time = datetime.datetime.now().strftime("%H:%M")
        speak(f"The time is {time}")
    elif "stop" in command:
        speak("Goodbye!")
        break
